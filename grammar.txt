stmt            -> FN|ACT IDENTIFIER LPAR (IDENTIFIER,)* RPAR LBRACE (expr|return|assignment)+ RBRACE SEMICOLON #Function/Non-returning function definition
                -> DEC ACT? IDENTIFIER LPAR (IDENTIFIER,)* RPAR SEMICOLON #Function declaration/prototype
                -> assignment #Only let assignment

return          -> RET expr

assignment      -> IDENTIFIER ("#") expr
                -> LET IDENTIFIER ("#") expr

expr            -> comparison

comparison      -> term ((">"|"<"|"=") term)*

term            -> factor (("+"|"-") factor)*

factor          -> unary ("*" unary)*

unary           -> ("+"|"-") unary 
                -> primary

primary         -> (NUMBER|IDENTIFIER) 
                -> LPAR expr RPAR
                -> IDENTIFIER LPAR (expr,)* RPAR #Function call
